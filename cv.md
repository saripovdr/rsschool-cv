5. Convert string to camel case

```
function toCamelCase(str) {
  let i = 0;
  let userStr = "";
  while (i < str.length) {
    if ((str[i] == "-") || (str[i] == "_")) {  
      userStr += str[++i].toUpperCase();
    } else {
        userStr += str[i]; 
    }
     i++; 
  }
  return userStr;
}
```
